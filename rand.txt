# Define the file path
$filePath = "C:\Path\To\Your\File.docx"

# Check if the file exists
if (Test-Path $filePath) {
    # Use Shell.Application COM object to access file metadata
    $shell = New-Object -ComObject Shell.Application
    $folder = $shell.Namespace((Get-Item $filePath).DirectoryName)
    $file = $folder.ParseName((Get-Item $filePath).Name)
    
    # Retrieve the "Last Saved By" property by its index
    $lastSavedBy = $folder.GetDetailsOf($file, 20)  # Index may vary, usually 20 in Office files
    
    if ($lastSavedBy) {
        Write-Output "Last saved by: $lastSavedBy"
    } else {
        Write-Output "The file does not have a 'Last Saved By' property."
    }
} else {
    Write-Output "File not found at the specified path."
}
